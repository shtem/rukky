   function primeInterval(l, u) displays all prime numbers between an interval [l,u)
   Input l = 900, u = 1000
   Output 907 911 919 929 937 941 947 953 967 971 977 983 991 997
$


:: void primeInterval := (real lower, real upper) {
    bool primeFlag

    for:: num := lower ; upper + 1 ; 1 {
        if:: num > 1 {
            primeFlag := true

            for:: i := 2 ; num ; 1 {
                if:: num % i <> 0 {
                    primeFlag := false
                    break::
                }
            }

            if:: primeFlag {
                display:num::
            }
            
        }
    }
}

real l := 900
real u := 1000
